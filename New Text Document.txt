def isBlack(im, trav):
    if im[trav.whereRU()[0], trav.whereRU()[1]][0] < 50 and im[trav.whereRU()[0], trav.whereRU()[1]][1] < 50 and im[trav.whereRU()[0], trav.whereRU()[1]][2] < 50:
        return True
    else:
        return False

def Paint(im, map):
    for i in range(0, im.shape[0] - 1):
        for j in range(0, im.shape[1] - 1):
            if map[i][j].all() == 0:
                im[i, j] = (0, 0, 255)
    return im
